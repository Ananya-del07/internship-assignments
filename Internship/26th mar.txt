# simple sentiment analyzer

positive_words = ["good", "great", "excellent", "amazing", "interesting", "enjoyable", "love"]
negative_words = ["bad", "boring", "poor", "slow", "hate", "worst", "not"]

reviews = [
    "The movie was very interesting and enjoyable",
    "I did not like the movie it was boring",
    "The story was amazing and the acting was very good",
    "The movie was too slow and not interesting",
    "Excellent movie with great performance"
]

for review in reviews:
    score = 0
    words = review.lower().split()
    
    for word in words:
        if word in positive_words:
            score += 1
        elif word in negative_words:
            score -= 1
    
    if score > 0:
        print(review, "-> Positive")
    else:
        print(review, "-> Negative")

Output

The movie was very interesting and enjoyable → Positive

I did not like the movie it was boring → Negative

The story was amazing and the acting was very good → Positive

The movie was too slow and not interesting → Negative

Excellent movie with great performance → Positive
